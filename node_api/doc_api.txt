// ** SESSION //
 end-point : '/sessions/:session_id' :

 	POST : Créer une session
 		Schema_Format :

 	GET: retourne la session avec l'id ':session_id' 
 	PUT : modifie la session avec l'id ':session_id' 
 	DELETE: Supprime la session avec l'id ':session_id'
